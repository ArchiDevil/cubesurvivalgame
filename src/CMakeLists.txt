cmake_minimum_required(VERSION 2.6)

if(MSVC)
	add_definitions(-DUNICODE -D_UNICODE -DD3D10_RENDER)
    include_directories($ENV{DXSDK_DIR}/include)
else()
	add_definitions(-std=c++0x)
endif()

add_subdirectory(GraphicsEngine)
add_subdirectory(GUI)
add_subdirectory(MathLib)
add_subdirectory(Utilities)
add_subdirectory(SimplePhysicsEngine)
add_subdirectory(game)

set_property(TARGET GraphicsEngine 		PROPERTY FOLDER "Libs")
set_property(TARGET GUI 				PROPERTY FOLDER "Libs")
set_property(TARGET MathLib 			PROPERTY FOLDER "Libs")
set_property(TARGET Utilities 			PROPERTY FOLDER "Libs")
set_property(TARGET SimplePhysicsEngine PROPERTY FOLDER "Libs")
set_property(TARGET Game 				PROPERTY FOLDER "Apps")

option(BUILD_TOOLS "Build engine tools" YES)

if(BUILD_TOOLS)
	add_subdirectory(Converter)
	add_subdirectory(MeshEditor)
	set_property(TARGET Converter  PROPERTY FOLDER "Apps")
	set_property(TARGET MeshEditor PROPERTY FOLDER "Apps")
endif()
